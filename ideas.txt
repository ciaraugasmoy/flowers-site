var width = window.innerWidth
var height = window.innerHeight
var radius = 12
var size = radius * 1 + 'px'

var el = d3.select('body').append('div')

var x = 0
var y = 0
var isOdd = false
while (true) {
    if (y > height) {
        break
    }

    el.append('span')
        .style('left', x + 'px')
        .style('top', y + 'px')
        .style('width', size)
        .style('height', size)
        .on('mouseenter', function() {
            d3.select(this)
                .classed('popped', true)
                .on('mouseenter', null)
        })

    x += radius * 0.8
    if (x > width) {
        x = isOdd ? 0 : radius
        isOdd = !isOdd
        y += radius * 0.8
    }
}
